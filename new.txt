import { useEffect, useRef } from 'react';
import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { Cake, Sparkles } from 'lucide-react';

gsap.registerPlugin(ScrollTrigger);

export const HeroSection = () => {
  const heroRef = useRef<HTMLDivElement>(null);
  const titleRef = useRef<HTMLHeadingElement>(null);
  const subtitleRef = useRef<HTMLParagraphElement>(null);
  const cakeRef = useRef<HTMLDivElement>(null);
  const contentRef = useRef<HTMLDivElement>(null);
  const imageAreaRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const tl = gsap.timeline();

    // Initial entrance animations
    tl.from(heroRef.current, {
      opacity: 0,
      scale: 0.8,
      duration: 0.8,
      ease: 'back.out',
    })
      .from(titleRef.current, {
        opacity: 0,
        y: 50,
        duration: 0.6,
        ease: 'power3.out',
      })
      .from(subtitleRef.current, {
        opacity: 0,
        y: 30,
        duration: 0.6,
        ease: 'power3.out',
      }, '-=0.3')
      .from(cakeRef.current, {
        opacity: 0,
        scale: 0,
        rotation: -180,
        duration: 0.8,
        ease: 'elastic.out',
      }, '-=0.4');

    // Continuous sparkle animation
    gsap.to('.sparkle', {
      rotation: 360,
      duration: 4,
      ease: 'linear',
      repeat: -1,
    });

    // Horizontal scroll animation - pin the section and slide content
    ScrollTrigger.create({
      trigger: heroRef.current,
      start: 'top top',
      end: '+=100%',
      pin: true,
      scrub: 1,
      onUpdate: (self) => {
        // Move content to the left
        gsap.to(contentRef.current, {
          x: -window.innerWidth * 0.5 * self.progress,
          opacity: 1 - (self.progress * 0.5),
          ease: 'none',
        });

        // Reveal image area from the right
        gsap.to(imageAreaRef.current, {
          opacity: self.progress,
          ease: 'none',
        });
      },
    });

    return () => {
      ScrollTrigger.getAll().forEach(trigger => trigger.kill());
    };
  }, []);

  return (
    <section ref={heroRef} className="min-h-screen flex items-center justify-center bg-gradient-hero relative overflow-hidden">
      <div className="absolute inset-0 bg-gradient-glow" />
      
      {/* Bottom fade overlay - creates smooth transition */}
      <div className="absolute bottom-0 left-0 right-0 h-32 bg-gradient-to-t from-background to-transparent z-30 pointer-events-none" />
      
      {/* Text Content - will slide left */}
      <div ref={contentRef} className="relative z-10 text-center max-w-4xl mx-auto px-4">
        <div className="mb-8 flex justify-center gap-8">
          <Sparkles className="sparkle w-12 h-12 text-primary" />
          <div ref={cakeRef}>
            <Cake className="w-24 h-24 text-secondary" fill="currentColor" />
          </div>
          <Sparkles className="sparkle w-12 h-12 text-accent" />
        </div>

        <h1
          ref={titleRef}
          className="text-6xl md:text-8xl font-bold mb-6 bg-gradient-celebration bg-clip-text text-transparent"
        >
          ðŸŽ‰ Happy Birthday! ðŸŽ‰
        </h1>

        <p
          ref={subtitleRef}
          className="text-xl md:text-2xl text-foreground/80 max-w-2xl mx-auto leading-relaxed"
        >
          Today is your special day, and we're here to celebrate the amazing person you are!
        </p>

        <div className="mt-12 inline-block">
          <div className="animate-bounce text-4xl">ðŸ‘‡</div>
          <p className="text-sm text-muted-foreground mt-2">Scroll down for a special message</p>
        </div>
      </div>

      {/* Image Area - will appear from right */}
      <div 
        ref={imageAreaRef}
        className="absolute right-0 top-0 w-1/2 h-full flex items-center justify-center opacity-0 z-20 px-8"
      >
        <div className="w-full h-full max-w-2xl max-h-[80vh] bg-white/10 backdrop-blur-sm rounded-2xl border-2 border-white/20 flex items-center justify-center">
          <p className="text-2xl text-white/60">Your image will go here</p>
        </div>
      </div>
    </section>
  );
};
